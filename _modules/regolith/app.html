    except (KeyError, AttributeError):
        abort(404)
    status = status_id = None
    if request.method == 'POST':
        form = request.form

            return shutdown()

            body = json_util.loads(form['body'].strip())
            status = 'canceled'
            status_id = str(body['_id'])
        elif 'save' in form:
            body = json_util.loads(form['body'].strip())
            coll.save(body) 
            status = 'saved ✓'
            status_id = str(body['_id'])
        elif 'add' in form:
            body = json_util.loads(form['body'].strip())
            added = insert_one(coll, body)
            status = 'added ✓'
            status_id = str(added.inserted_id)
        elif 'delete' in form:
